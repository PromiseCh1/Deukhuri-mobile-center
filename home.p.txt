<script>
// Filter visible grid (only affects the visible products in #visibleGrid)
const filterBtns = document.querySelectorAll('.filter-btn');
const visibleCards = document.querySelectorAll('#visibleGrid .product-card');
filterBtns.forEach(btn => {
  btn.addEventListener('click', () => {
    const filter = btn.dataset.filter;
    filterBtns.forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    visibleCards.forEach(card => {
      const category = card.dataset.category;
      if (filter === 'all' || category === filter) {
        card.style.display = 'block';
      } else {
        card.style.display = 'none';
      }
    });
  });
});

// Toggle horizontal scroll (View More / View Less)
const viewMoreBtn = document.getElementById('viewMoreBtn');
const horizontalGrid = document.getElementById('horizontalGrid');
if (viewMoreBtn && horizontalGrid) {
  viewMoreBtn.addEventListener('click', () => {
    horizontalGrid.classList.toggle('show');
    if (horizontalGrid.classList.contains('show')) {
      viewMoreBtn.textContent = 'View Less ↑';
      horizontalGrid.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    } else {
      viewMoreBtn.textContent = 'View More →';
    }
  });
}

// Modal logic
const modal = document.getElementById('productModal');
const modalImg = document.getElementById('modalImage');
const modalTitle = document.getElementById('modalTitle');
const modalPrice = document.getElementById('modalPrice');
const whatsappBtn = document.getElementById('whatsappBtn');
let currentProduct = {};

function openModal(id, img, name, price) {
  modalImg.src = img;
  modalTitle.innerText = name;
  modalPrice.innerText = price;
  currentProduct = { id, name, price };
  modal.style.display = 'flex';
  const msg = `Hello, I want to purchase ${name} (${price}). My FreeFire/PUBG UID is: _______. I will send Esewa payment screenshot.`;
  whatsappBtn.href = `https://wa.me/9847956550?text=${encodeURIComponent(msg)}`;
}

// Attach click to all product cards (both grids)
document.querySelectorAll('.product-card').forEach(card => {
  card.addEventListener('click', (e) => {
    const img = card.querySelector('.product-img img').src;
    const name = card.querySelector('.product-info h3').innerText;
    const priceText = card.querySelector('.price').innerText;
    const productId = card.dataset.productId;
    openModal(productId, img, name, priceText);
  });
});

// Close modal
document.querySelector('.modal-close')?.addEventListener('click', () => modal.style.display = 'none');
window.addEventListener('click', (e) => { if (e.target === modal) modal.style.display = 'none'; });
</script>
